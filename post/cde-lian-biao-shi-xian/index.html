C++的链表实现

2024-03-30 2 min read # 数据结构

链表是一种线性表，在插入数据和删除数据的时候，相比于数组更加的快捷。下面是链表的实现。

#include <iostream>

// 定义链表类
class LinkedList {
private:
    class ListNode {
    public:
        int data; // 存储的数据
        ListNode* next; // 指向下一个节点的指针
        ListNode(int value): data(value), next(nullptr) {};
};
public:
    ListNode* head; // 链表的头节点
    // 构造函数
    LinkedList() : head(nullptr) {}
    // 向链表中添加元素
    void addValue(int value) {
        ListNode* current = head;
        if(head == nullptr) {
            head = new ListNode(value);
        }
        else {
            while(current->next!=nullptr) current = current->next;
            current->next = new ListNode(value);
        }
    }
    // 打印链表
    void printList() {
        ListNode* current = head;
        while(current != nullptr){
            std::cout << current->data << " ";
            current = current->next;
        }
        std::cout<<'\n';
    }
    // 删除指定位置节点
    void removeValue(int index){
        ListNode* current = head;
        ListNode* previous = nullptr;
        for(int i = 0; i < index && current != nullptr; ++i){
            previous = current;
            current = current->next;
        }
        if(current != nullptr){
            if(previous == nullptr){
                head = current->next;
            }
            else {
                previous->next = current->next;
            }
            delete current;
        }
    }
    // 修改指定节点元素
    void reviseValue(int index, int value){
        ListNode* current = head;
        for(int i = 0; i < index && current != nullptr; ++i){
            current = current->next;
        }
        if(current == nullptr) delete current;
        else current->data = value;
    }
    // 插入指定位置的元素
    void insertValue(int index, int value){
        ListNode* current = head;
        ListNode* node = new ListNode(value);
        if(head != nullptr){
            if(index == 0){
                node->next = head;
                head = node;
            }
            else{
                while(current!=nullptr && --index){
                    current = current->next;
                }
                if(current !=nullptr && index == 0){
                    node->next = current->next;
                    current->next = node;
                }
            }
        }
        else{
            if(index == 0){
                head = new ListNode(value);
            }
        }
}

    // 析构函数
    ~LinkedList() {
        ListNode* current = head;
        while (current != nullptr) {
            ListNode* next = current->next;
            delete current;
            current = next;
        }
    }
};
