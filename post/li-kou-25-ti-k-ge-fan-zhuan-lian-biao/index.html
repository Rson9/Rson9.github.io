力扣25题-K个反转链表

2025-04-21 2 min read

问题拆分

由于是每K个位置，进行一次反转，所以我们需要先设计指定head到指定tail的反转，返回新的链表指向。
同时我们需要构造一个虚拟节点hair，用于最后返回新的链表指向。
同时，在遍历中，每次记录上次反转的最后一个节点为prev和当前tail的下一个节点next，来处理当前反转之后的前置链接和后置链接。最后返回hair.Next就行了。

/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reverseKGroup(head *ListNode, k int) *ListNode {
    hair := &ListNode{Next:head}

    prev := hair

    for head!=nil{
        tail := prev
        for i:=1;i<=k;i++{
            tail = tail.Next
            if tail == nil{
                return hair.Next
            }
        }
        nex := tail.Next
        head,tail = MyReverse(head,tail)
        prev.Next=head
        tail.Next = nex
        prev = tail
        head = nex
    }
    return hair.Next
}

func MyReverse(head,tail *ListNode)(*ListNode,*ListNode){
    prev := tail.Next
    p:=head
    for prev !=tail{
        next:=p.Next
        p.Next = prev
        prev = p
        p = next
    }
    return tail,head
}
